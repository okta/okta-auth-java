241     private void handleResult(AuthenticationResponse authenticationResponse, AuthenticationStateHandler authenticationStateHandler) {
242         AuthenticationStatus status = authenticationResponse.getStatus();
243 
244          switch (status) {
245              case SUCCESS:
246                  authenticationStateHandler.handleSuccess(authenticationResponse);
247                  break;
248              case LOCKED_OUT:
249                  authenticationStateHandler.handleLockedOut(authenticationResponse);
250                  break;
251              case MFA_CHALLENGE:
252                  authenticationStateHandler.handleMfaChallenge(authenticationResponse);
253                  break;
254              case MFA_ENROLL:
255                  authenticationStateHandler.handleMfaEnroll(authenticationResponse);
256                  break;
257              case MFA_ENROLL_ACTIVATE:
258                  authenticationStateHandler.handleMfaEnrollActivate(authenticationResponse);
259                  break;
260              case MFA_REQUIRED:
261                  authenticationStateHandler.handleMfaRequired(authenticationResponse);
262                  break;
263              case PASSWORD_EXPIRED:
264                  authenticationStateHandler.handlePasswordExpired(authenticationResponse);
265                  break;
266              case PASSWORD_RESET:
267                  authenticationStateHandler.handlePasswordReset(authenticationResponse);
268                  break;
269              case PASSWORD_WARN:
270                  authenticationStateHandler.handlePasswordWarning(authenticationResponse);
271                  break;
272              case RECOVERY:
273                  authenticationStateHandler.handleRecovery(authenticationResponse);
274                  break;
275              case RECOVERY_CHALLENGE:
276                  authenticationStateHandler.handleRecoveryChallenge(authenticationResponse);
277                  break;
278              case UNAUTHENTICATED:
279                  authenticationStateHandler.handleUnauthenticated(authenticationResponse);
280                  break;
281              default:
282                  authenticationStateHandler.handleUnknown(authenticationResponse);
283          }
284     }

286     private AuthenticationResponse doPost(String href, Resource request, AuthenticationStateHandler authenticationStateHandler, RequestContext requestContext) throws AuthenticationException {
287         try {
288 
289             Map<String, Object> query = null;
290             Map<String, List<String>> headers = null;
291 
292             if (requestContext != null) {
293                 query = requestContext.getQueryParams().stream()
294                         .collect(Collectors.toMap(QueryParameter::getKey, QueryParameter::getValue));
295 
296                 headers = requestContext.getHeaders().stream()
297                         .collect(Collectors.toMap(Header::getKey, Header::getValue, (a, b) -> Stream.concat(a.stream(), b.stream()).collect(Collectors.toList())));
298             }
299 
300             AuthenticationResponse authenticationResponse = getDataStore().create(href, request, null, AuthenticationResponse.class, query, headers);
301 
302             if (authenticationStateHandler != null) {
303                 handleResult(authenticationResponse, authenticationStateHandler);
304             }
305 
306             return authenticationResponse;
307         } catch (ResourceException e) {
308             translateException(e);
309             throw e; // above method should always throw
310         }
311     }
312 
313     private void translateException(ResourceException resourceException) throws AuthenticationException {
314 
315         String errorCode = resourceException.getCode();
316 
317         switch (errorCode) {
318             case AuthenticationFailureException.ERROR_CODE:
319                 throw new AuthenticationFailureException(resourceException);
320 
321             case CredentialsException.ERROR_CODE:
322                 throw new CredentialsException(resourceException);
323 
324             case FactorValidationException.ERROR_CODE:
325                 throw new FactorValidationException(resourceException);
326 
327             case InvalidAuthenticationStateException.ERROR_CODE:
328                 throw new InvalidAuthenticationStateException(resourceException);
329 
330             case InvalidRecoveryAnswerException.ERROR_CODE:
331                 throw new InvalidRecoveryAnswerException(resourceException);
332 
333             case InvalidTokenException.ERROR_CODE:
334                 throw new InvalidTokenException(resourceException);
335 
336             case InvalidUserException.ERROR_CODE:
337                 throw new InvalidUserException(resourceException);
338 
339             default:
340                 throw resourceException;
341         }
