39      private static final String PREVIOUS_AUTHN_RESULT = "previousAuthNResult";

41      @Override
42      public void handleSuccess(AuthenticationResponse successResponse) {
43          // the last request was a success, but if we do not have a session token
44          // we need to force the flow to start over
45          String relayState = successResponse.getRelayState();
46          String dest = relayState != null ? relayState : "/troopers";
47          redirect(dest, successResponse);
48      }
49  
50      @Override
51      public void handlePasswordExpired(AuthenticationResponse passwordExpired) {
52          redirect("/login/change-password", passwordExpired);
53      }
54  
55      @Override
56      public void handleMfaRequired(AuthenticationResponse mfaRequiredResponse) {
57          redirect("/login/mfa", mfaRequiredResponse);
58      }
59  
60      @Override
61      public void handleMfaChallenge(AuthenticationResponse mfaChallengeResponse) {
62          // do nothing this will be handled by the caller
63          String factorType = mfaChallengeResponse.getFactors().get(0).getType().name().toLowerCase(Locale.ENGLISH);
64          redirect("/login/mfa/verify/"+ factorType, mfaChallengeResponse);
65      }
66  
67      @Override
68      public void handleLockedOut(AuthenticationResponse lockedOutResponse) {
69          redirect("/login/unlock", lockedOutResponse);
70      }
71  
72      @Override
73      public void handleRecoveryChallenge(AuthenticationResponse recoveryChallenge) {
74          redirect("/login/unlock/recovery", recoveryChallenge);
75      }
76  
77      @Override
78      public void handleRecovery(AuthenticationResponse recoveryResponse) {
79          redirect("/login/recovery", recoveryResponse);
80      }
81  
82      @Override
83      public void handleMfaEnroll(AuthenticationResponse mfaEnroll) {
84          redirect("/login/mfa/enroll", mfaEnroll);
85      }
86

88      public void handleMfaEnrollActivate(AuthenticationResponse mfaEnrollActivate) {
89          String factorType = mfaEnrollActivate.getFactors().get(0).getType().name().toLowerCase(Locale.ENGLISH);
90          redirect("/login/mfa/activate/"+ factorType, mfaEnrollActivate);

93      @Override
94      public void handlePasswordReset(AuthenticationResponse passwordReset) {
95          redirect("/login/reset", passwordReset);
96      }
97  
98      @Override
99      public void handleUnknown(AuthenticationResponse unknownResponse) {
100         redirect("/login?error="+ unknownResponse.getStatus().name(), unknownResponse);
101     }

103     private void log(AuthenticationResponse authenticationResponse) {
104          String pid = ManagementFactory.getRuntimeMXBean().getName();
105         File targetDir = new File("target/"+pid);
106         targetDir.mkdirs();
107         File outFile = new File(targetDir, "response-"+System.currentTimeMillis()+".json");
108 
109         ReferenceFactory referenceFactory = new ReferenceFactory();
110         DefaultResourceConverter resourceConverter = new DefaultResourceConverter(referenceFactory);
111         Assert.isInstanceOf(AbstractResource.class, authenticationResponse);
112         Map<String, Object> data = resourceConverter.convert((AbstractResource) authenticationResponse, false);
113 
114         JacksonMapMarshaller mapMarshaller = new JacksonMapMarshaller();
115         try(FileOutputStream fos = new FileOutputStream(outFile)) {
116             mapMarshaller.marshal(fos, data);
117         } catch (IOException e) {
118             throw new IllegalStateException("failed to serialize response.", e);
119         }
120     }
121 
122     private void redirect(String location, AuthenticationResponse authenticationResponse) {
123 
124         log(authenticationResponse);
125         try {
126             Subject subject = SecurityUtils.getSubject();
127             HttpServletResponse response = WebUtils.getHttpResponse(subject);
128             setAuthNResult(authenticationResponse);
129             response.sendRedirect(location);
130         } catch (IOException e) {
131             throw new IllegalStateException("failed to redirect.", e);
132         }
133     }
134 
135     static void setAuthNResult(AuthenticationResponse authenticationResponse) {
136         Subject subject = SecurityUtils.getSubject();
137         subject.getSession().setAttribute(PREVIOUS_AUTHN_RESULT, authenticationResponse);
138     }
