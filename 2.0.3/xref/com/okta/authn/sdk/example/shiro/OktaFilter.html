16  package com.okta.authn.sdk.example.shiro;
17  
18  import com.okta.authn.sdk.example.ExampleAuthenticationStateHandler;
19  import com.okta.authn.sdk.resource.AuthenticationResponse;
20  import com.okta.authn.sdk.resource.AuthenticationStatus;
21  import org.apache.shiro.authc.AuthenticationToken;
22  import org.apache.shiro.web.filter.authc.AuthenticatingFilter;
23  
24  import javax.servlet.ServletRequest;
25  import javax.servlet.ServletResponse;
26  
27  public class OktaFilter extends AuthenticatingFilter {
28  
29      public OktaFilter() {
30          this.appliedPaths.put(getLoginUrl(), null);
31      }
32  
33      @Override
34      protected AuthenticationToken createToken(ServletRequest request, ServletResponse response) {
35          return new OktaSuccessLoginToken(ExampleAuthenticationStateHandler.getPreviousAuthResult());
36      }
37  
38      @Override
39      protected boolean onAccessDenied(ServletRequest request, ServletResponse response) throws Exception {
40  
41          if (isLoginRequest(request, response)) {
42              return executeLogin(request, response);
43          }
44  
45          if (pathsMatch(getLoginUrl(), request)) {
46              return true;
47          }
48  
49          saveRequestAndRedirectToLogin(request, response);
50          return false;
51      }
52  
53      @Override
54      public boolean isLoginRequest(ServletRequest request, ServletResponse response) {
55  
56          AuthenticationResponse result = ExampleAuthenticationStateHandler.getPreviousAuthResult();
57          return result != null
58                  && result.getStatus().equals(AuthenticationStatus.SUCCESS)
59                  && result.getSessionToken() != null;
60      }
61  }
