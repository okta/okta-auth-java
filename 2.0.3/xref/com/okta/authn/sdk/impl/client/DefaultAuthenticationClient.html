247     private void handleResult(AuthenticationResponse authenticationResponse, AuthenticationStateHandler authenticationStateHandler) {
248         AuthenticationStatus status = authenticationResponse.getStatus();
249 
250          switch (status) {
251              case SUCCESS:
252                  authenticationStateHandler.handleSuccess(authenticationResponse);
253                  break;
254              case LOCKED_OUT:
255                  authenticationStateHandler.handleLockedOut(authenticationResponse);
256                  break;
257              case MFA_CHALLENGE:
258                  authenticationStateHandler.handleMfaChallenge(authenticationResponse);
259                  break;
260              case MFA_ENROLL:
261                  authenticationStateHandler.handleMfaEnroll(authenticationResponse);
262                  break;
263              case MFA_ENROLL_ACTIVATE:
264                  authenticationStateHandler.handleMfaEnrollActivate(authenticationResponse);
265                  break;
266              case MFA_REQUIRED:
267                  authenticationStateHandler.handleMfaRequired(authenticationResponse);
268                  break;
269              case PASSWORD_EXPIRED:
270                  authenticationStateHandler.handlePasswordExpired(authenticationResponse);
271                  break;
272              case PASSWORD_RESET:
273                  authenticationStateHandler.handlePasswordReset(authenticationResponse);
274                  break;
275              case PASSWORD_WARN:
276                  authenticationStateHandler.handlePasswordWarning(authenticationResponse);
277                  break;
278              case RECOVERY:
279                  authenticationStateHandler.handleRecovery(authenticationResponse);
280                  break;
281              case RECOVERY_CHALLENGE:
282                  authenticationStateHandler.handleRecoveryChallenge(authenticationResponse);
283                  break;
284              case UNAUTHENTICATED:
285                  authenticationStateHandler.handleUnauthenticated(authenticationResponse);
286                  break;
287              default:
288                  authenticationStateHandler.handleUnknown(authenticationResponse);
289          }
290     }

292     private AuthenticationResponse doPost(String href, Resource request, AuthenticationStateHandler authenticationStateHandler, RequestContext requestContext) throws AuthenticationException {
293         try {
294 
295             Map<String, Object> query = null;
296             Map<String, List<String>> headers = null;
297 
298             if (requestContext != null) {
299                 query = requestContext.getQueryParams().stream()
300                         .collect(Collectors.toMap(QueryParameter::getKey, QueryParameter::getValue));
301 
302                 headers = requestContext.getHeaders().stream()
303                         .collect(Collectors.toMap(Header::getKey, Header::getValue, (a, b) -> Stream.concat(a.stream(), b.stream()).collect(Collectors.toList())));
304             }
305 
306             AuthenticationResponse authenticationResponse = getDataStore().create(href, request, null, AuthenticationResponse.class, query, headers);
307 
308             if (authenticationStateHandler != null) {
309                 handleResult(authenticationResponse, authenticationStateHandler);
310             }
311 
312             return authenticationResponse;
313         } catch (ResourceException e) {
314             translateException(e);
315             throw e; // above method should always throw
316         }
317     }

319     private void translateException(ResourceException resourceException) throws AuthenticationException {
320 
321         String errorCode = resourceException.getCode();
322 
323         switch (errorCode) {
324             case AuthenticationFailureException.ERROR_CODE:
325                 throw new AuthenticationFailureException(resourceException);
326 
327             case CredentialsException.ERROR_CODE:
328                 throw new CredentialsException(resourceException);
329 
330             case FactorValidationException.ERROR_CODE:
331                 throw new FactorValidationException(resourceException);
332 
333             case InvalidAuthenticationStateException.ERROR_CODE:
334                 throw new InvalidAuthenticationStateException(resourceException);
335 
336             case InvalidRecoveryAnswerException.ERROR_CODE:
337                 throw new InvalidRecoveryAnswerException(resourceException);
338 
339             case InvalidTokenException.ERROR_CODE:
340                 throw new InvalidTokenException(resourceException);
341 
342             case InvalidUserException.ERROR_CODE:
343                 throw new InvalidUserException(resourceException);
344 
345             case UserLockedException.ERROR_CODE:
346                 throw new UserLockedException(resourceException);
347 
348             default:
349                 throw resourceException;
350         }
