112     @GET
113     @Path("/mfa/enroll")
114     public MfaEnrollSelectionView getMfaEnrollSelectionView() {
115 
116         List<Factor> factors = getPreviousAuthResult().getFactors().stream()
117             .map(authFactor -> {
118                     String shortType = MfaVerifyView.relativeLink(authFactor);
119                     return new Factor(authFactor.getId(),
120                                       shortType,
121                                       authFactor.getProvider().name(),
122                                       authFactor.getVendorName(),
123                                       authFactor.getProfile(),
124                                       "/login/mfa/enroll/" + shortType);})
125             .collect(Collectors.toList());
126         return new MfaEnrollSelectionView(factors);
127     }

160     @GET
161     @Path("/mfa/verify/{type}")
162     public MfaVerifyView getVerifyMfaView(@PathParam("type") String type) throws AuthenticationException {
163 
164         AuthenticationResponse authenticationResponse = getPreviousAuthResult();
165         com.okta.authn.sdk.resource.Factor factor = getFactor(type, authenticationResponse);
166 
167         if (factor.getType().equals(FactorType.TOKEN_SOFTWARE_TOTP)) {
168             return new MfaVerifyView(factor);
169         } else {
170             return new MfaVerifyView(challengeFactor(factor, authenticationResponse));
171         }
172     }

183     @POST
184     public Response doLogin(@FormParam("username") String username,
185                             @FormParam("password") String password) throws AuthenticationException {
186 
187         char[] pass = password != null ? password.toCharArray() : null;
188 
189         try {
190             authenticationClient.authenticate(username, pass, null, new ExampleAuthenticationStateHandler());
191             // the state handler will redirect
192         } catch (AuthenticationFailureException e) {
193             return Response.ok(new LoginView(Optional.of(e))).build();
194         }
195         return null;
196     }

320     @POST

322     public void enrollMfa(@PathParam("factorType") String factorType, Form form) throws AuthenticationException {
323         FactorEnrollRequest request = authenticationClient.instantiate(FactorEnrollRequest.class)
324                 .setProvider(FactorProvider.OKTA)
325                 .setStateToken(getPreviousAuthResult().getStateToken())
326                 .setFactorType(MfaVerifyView.fromRelativeLink(factorType))
327                 .setFactorProfile(authenticationClient.instantiate(SmsFactorProfile.class)

329         authenticationClient.enrollFactor(request, new ExampleAuthenticationStateHandler());
330     }

332     private com.okta.authn.sdk.resource.Factor getFactor(String type, AuthenticationResponse authenticationResponse) {
333 
334         FactorType oktaType = MfaVerifyView.fromRelativeLink(type);
335         return authenticationResponse.getFactors().stream()
336                 .filter(it -> it.getType().equals(oktaType))
337                 .findFirst().get();
338     }
339 
340     private com.okta.authn.sdk.resource.Factor challengeFactor(com.okta.authn.sdk.resource.Factor factor, AuthenticationResponse authenticationResponse) throws AuthenticationException {
341 
342         AuthenticationResponse challengeResult = authenticationClient.challengeFactor(factor.getId(), authenticationResponse.getStateToken(), new ExampleAuthenticationStateHandler());
343 
344         // check the response type
345         if (!challengeResult.getStatus().equals(AuthenticationStatus.MFA_CHALLENGE)) {
346             throw new IllegalStateException("Challenge Result is empty, and other state was not triggered");
347         }
348 
349         return challengeResult.getFactors().get(0);
350     }
